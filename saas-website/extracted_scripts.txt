<script>
      let currentUser = null;
      document.addEventListener("DOMContentLoaded", function () {
        loadDashboard();
      });
      function loadDashboard() {
        const userEmail = localStorage.getItem("userEmail");
        const signupDate = localStorage.getItem("signupDate");
        if (!userEmail) {
          window.location.href = "index.html";
          return;
        }
        currentUser = {
          email: userEmail,
          created_at: signupDate || new Date().toISOString(),
          message_count: 0,
          message_limit: 300,
          widget_status: "none",
        };
        document.getElementById("user-email").textContent = currentUser.email;
        updateDashboardStats();
        updateWidgetSection();
        updateQuickActions();
        document.getElementById("loading-screen").style.display = "none";
        document.getElementById("dashboard-content").style.display = "block";
      }
      function updateDashboardStats() {
        document.getElementById("message-count").textContent =
          currentUser.message_count;
        document.getElementById("widget-count").textContent =
          currentUser.widget_status === "none" ? 0 : 1;
        const remainingMessages =
          currentUser.message_limit - currentUser.message_count;
        const percentage = Math.round(
          (remainingMessages / currentUser.message_limit) * 100,
        );
        document.getElementById("messages-remaining").textContent =
          percentage + "%";
        const createdDate = new Date(currentUser.created_at);
        const daysSince = Math.floor(
          (new Date() - createdDate) / (1000 * 60 * 60 * 24),
        );
        document.getElementById("days-active").textContent = daysSince;
      }
      function updateWidgetSection() {
        const widgetSection = document.getElementById("widget-section");
        if (currentUser.widget_status === "none") {
          widgetSection.innerHTML =
            '<div class="no-widget"><div class="no-widget-icon">🤖</div><h4>Create Your First Chatbot</h4><p>Upload your PDF documents and create a custom AI chatbot that can answer questions about your content. Perfect for customer support, documentation, or any knowledge base.</p><button class="create-widget-btn" onclick="startWidgetCreation()">🚀 Create Your Chatbot</button></div>';
        } else {
          const customization = JSON.parse(
            localStorage.getItem("widgetCustomization") || "{}",
          );
          widgetSection.innerHTML =
            '<div class="widget-created"><div class="widget-status-badge status-active">Active</div><h4>' +
            (customization.chatTitle || "AI Assistant") +
            '</h4><p>Your custom AI chatbot is ready to use on your website.</p><div style="margin-top: 1.5rem;"><button class="btn btn-primary" onclick="viewEmbedCode()">📋 Get Embed Code</button><button class="btn btn-outline" onclick="openSettings()">⚙️ Settings</button></div></div>';
        }
      }
      function updateQuickActions() {
        const quickActions = document.getElementById("quick-actions");
        if (currentUser.widget_status === "none") {
          quickActions.innerHTML =
            '<button class="btn btn-primary" onclick="startWidgetCreation()">🚀 Create Your Chatbot</button><button class="btn btn-outline" onclick="viewDocumentation()">📖 View Documentation</button><button class="btn btn-outline" onclick="viewExamples()">👀 See Examples</button><button class="btn btn-outline" onclick="contactSupport()">💬 Contact Support</button>';
        } else {
          quickActions.innerHTML =
            '<button class="btn btn-primary" onclick="viewEmbedCode()">📋 Get Embed Code</button><button class="btn btn-outline" onclick="openSettings()">⚙️ Widget Settings</button><button class="btn btn-outline" onclick="viewAnalytics()">📊 View Analytics</button><button class="btn btn-outline" onclick="contactSupport()">💬 Contact Support</button>';
        }
      }
      // Modal and Upload Logic
      let selectedFiles = [];
      let processingResults = null;
      function startWidgetCreation() {
        selectedFiles = [];
        processingResults = null;
        [
          "step-upload",
          "step-processing-complete",
          "step-bubble-customization",
          "step-chat-customization",
          "step-embed-code",
        ].forEach((id, index) => {
          document.getElementById(id).style.display =
            index === 0 ? "block" : "none";
        });
        document.getElementById("modal-title").textContent =
          "🚀 Create Your Chatbot - Step 1 of 5";
        resetUploadForm();
        document.getElementById("widget-creation-modal").style.display = "flex";
      }
      function closeWidgetModal() {
        document.getElementById("widget-creation-modal").style.display = "none";
        resetUploadForm();
      }
      function resetUploadForm() {
        document.getElementById("pdf-file-input").value = "";
        document.getElementById("file-list").style.display = "none";
        document.getElementById("upload-progress").style.display = "none";
        document.getElementById("upload-next-btn").disabled = true;
        document.getElementById("files-container").innerHTML = "";
        selectedFiles = [];
      }
      function viewEmbedCode() {
        startWidgetCreation();
        goToEmbedCodeStep(true);
      }
      // Placeholder functions
      function openSettings() {
        alert("⚙️ Settings panel coming soon!");
      }
      function viewDocumentation() {
        alert("📖 Documentation coming soon!");
      }
      function viewExamples() {
        alert("👀 Examples gallery coming soon!");
      }
      function viewAnalytics() {
        alert("📊 Analytics dashboard coming soon!");
      }
      function contactSupport() {
        alert("💬 Support system coming soon!");
      }
      function openTestPage() {
        alert("🧪 Test page coming soon!");
      }

      // File Handling
      document.addEventListener("DOMContentLoaded", function () {
        const uploadArea = document.getElementById("pdf-upload-area");
        const fileInput = document.getElementById("pdf-file-input");
        if (uploadArea && fileInput) {
          ["dragover", "dragleave", "drop"].forEach((eventName) =>
            uploadArea.addEventListener(eventName, (e) => e.preventDefault()),
          );
          uploadArea.addEventListener("dragover", () =>
            uploadArea.classList.add("dragover"),
          );
          uploadArea.addEventListener("dragleave", () =>
            uploadArea.classList.remove("dragover"),
          );
          uploadArea.addEventListener("drop", (e) => {
            uploadArea.classList.remove("dragover");
            const files = Array.from(e.dataTransfer.files).filter(
              (f) => f.type === "application/pdf",
            );
            handleFileSelection(files);
          });
          uploadArea.addEventListener("click", () => fileInput.click());
          fileInput.addEventListener("change", (e) =>
            handleFileSelection(Array.from(e.target.files)),
          );
        }
        const modal = document.getElementById("widget-creation-modal");
        if (modal)
          modal.addEventListener(
            "click",
            (e) => e.target === modal && closeWidgetModal(),
          );
      });
      function handleFileSelection(files) {
        files.forEach((file) => {
          if (
            !selectedFiles.find(
              (f) => f.name === file.name && f.size === file.size,
            )
          ) {
            selectedFiles.push(file);
          }
        });
        displaySelectedFiles();
        document.getElementById("upload-next-btn").disabled =
          selectedFiles.length === 0;
      }
      function displaySelectedFiles() {
        const container = document.getElementById("files-container");
        const fileList = document.getElementById("file-list");
        fileList.style.display = selectedFiles.length > 0 ? "block" : "none";
        container.innerHTML = selectedFiles
          .map(
            (file, index) =>
              `<div class="file-item">
              <div class="file-info">
                <div class="file-icon">📄</div>
                <div class="file-details">
                  <h6>${file.name}</h6>
                  <p>${formatFileSize(file.size)} • PDF Document</p>
                </div>
              </div>
              <button class="file-remove" onclick="removeFile(${index})" title="Remove file">✕</button>
            </div>`,
          )
          .join("");
      }
      function removeFile(index) {
        selectedFiles.splice(index, 1);
        displaySelectedFiles();
        document.getElementById("upload-next-btn").disabled =
          selectedFiles.length === 0;
      }
      function formatFileSize(bytes) {
        if (bytes === 0) return "0 Bytes";
        const i = Math.floor(Math.log(bytes) / Math.log(1024));
        return `${parseFloat((bytes / Math.pow(1024, i)).toFixed(2))} ${
          ["Bytes", "KB", "MB", "GB"][i]
        }`;
      }

      // Document Processing
      async function processDocuments() {
        if (selectedFiles.length === 0) return;
        document.getElementById("upload-progress").style.display = "block";
        document.getElementById("upload-next-btn").disabled = true;

        try {
          updateProgress(
            20,
            "Uploading files...",
            "Preparing your documents for processing.",
          );
          await new Promise((resolve) => setTimeout(resolve, 1000));

          updateProgress(
            60,
            "Extracting text...",
            "Reading content from your PDF documents.",
          );
          await new Promise((resolve) => setTimeout(resolve, 1500));

          const mockResult = {
            message: "Documents processed successfully",
            chunks_count: Math.floor(Math.random() * 451) + 50,
            user: currentUser.email,
          };
          processingResults = mockResult;

          updateProgress(
            90,
            "Creating embeddings...",
            "Preparing your documents for intelligent search.",
          );
          await new Promise((resolve) => setTimeout(resolve, 1000));

          updateProgress(
            100,
            "Processing complete!",
            "Your documents are ready for the chatbot.",
          );
          await new Promise((resolve) => setTimeout(resolve, 500));

          showProcessingComplete();
        } catch (error) {
          console.error("Processing error:", error);
          updateProgress(
            0,
            "Processing failed",
            "An unexpected error occurred. Please try again.",
          );
          setTimeout(() => {
            document.getElementById("upload-next-btn").disabled = false;
            document.getElementById("upload-progress").style.display = "none";
          }, 3000);
        }
      }

      function updateProgress(percentage, status, details) {
        document.getElementById("progress-fill").style.width = `${percentage}%`;
        document.getElementById("progress-status").textContent = status;
        document.getElementById("progress-details").textContent = details;
      }

      // Navigation Functions
      function showProcessingComplete() {
        document.getElementById("step-upload").style.display = "none";
        document.getElementById("step-processing-complete").style.display =
          "block";
        document.getElementById("modal-title").textContent =
          "🚀 Create Your Chatbot - Step 2 of 5";
        const summary = document.getElementById("processing-summary");
        const totalFiles = selectedFiles.length;
        const totalSize = selectedFiles.reduce(
          (sum, file) => sum + file.size,
          0,
        );
        summary.innerHTML = `<div class="summary-item"><span>📄 Documents processed:</span><span>${totalFiles} PDF${
          totalFiles > 1 ? "s" : ""
        }</span></div>
           <div class="summary-item"><span>📊 Total size:</span><span>${formatFileSize(
             totalSize,
           )}</span></div>
           <div class="summary-item"><span>🧠 Text chunks created:</span><span>${
             processingResults?.chunks_count || 0
           }</span></div>
           <div class="summary-item"><span>✅ Status:</span><span style="color: #00a86b;">Ready for customization</span></div>`;
      }
      function goToCustomization() {
        document.getElementById("step-processing-complete").style.display =
          "none";
        document.getElementById("step-bubble-customization").style.display =
          "block";
        document.getElementById("modal-title").textContent =
          "🚀 Create Your Chatbot - Step 3 of 5";
        updateBubblePreview();
      }
      function goBackToProcessing() {
        document.getElementById("step-bubble-customization").style.display =
          "none";
        document.getElementById("step-processing-complete").style.display =
          "block";
        document.getElementById("modal-title").textContent =
          "🚀 Create Your Chatbot - Step 2 of 5";
      }
      function goToChatCustomization() {
        document.getElementById("step-bubble-customization").style.display =
          "none";
        document.getElementById("step-chat-customization").style.display =
          "block";
        document.getElementById("modal-title").textContent =
          "🚀 Create Your Chatbot - Step 4 of 5";
        updateChatPreview();
      }
      function goBackToBubbleCustomization() {
        document.getElementById("step-chat-customization").style.display =
          "none";
        document.getElementById("step-bubble-customization").style.display =
          "block";
        document.getElementById("modal-title").textContent =
          "🚀 Create Your Chatbot - Step 3 of 5";
      }
      function goToEmbedCodeStep(isExisting = false) {
        if (!isExisting) {
          document.getElementById("step-chat-customization").style.display =
            "none";
        } else {
          [
            "step-upload",
            "step-processing-complete",
            "step-bubble-customization",
            "step-chat-customization",
          ].forEach((id) => {
            document.getElementById(id).style.display = "none";
          });
        }
        document.getElementById("step-embed-code").style.display = "block";
        document.getElementById("modal-title").textContent =
          "🚀 Your Chatbot is Ready! - Step 5 of 5";
        generateAndDisplayEmbedCode();
      }
      function goBackToChatCustomization() {
        document.getElementById("step-embed-code").style.display = "none";
        document.getElementById("step-chat-customization").style.display =
          "block";
        document.getElementById("modal-title").textContent =
          "🚀 Create Your Chatbot - Step 4 of 5";
      }

      // Customization & Preview Functions
      function updateBubblePreview() {
        const bubbleColor = document.getElementById("bubble-color").value;
        const bubbleIcon = document.getElementById("bubble-icon").value;
        const bubblePosition = document.getElementById("bubble-position").value;
        const bubbleSize = document.getElementById("bubble-size").value;
        const bubbleText = document.getElementById("bubble-text").value;
        const enablePulse = document.getElementById("bubble-pulse").checked;
        const enableShadow = document.getElementById("bubble-shadow").checked;

        const bubble = document.getElementById("chat-bubble-preview");
        bubble.style.background = bubbleColor;
        let className = `chat-bubble-preview position-${bubblePosition} size-${bubbleSize}`;
        if (!enablePulse) className += " no-pulse";
        if (!enableShadow) className += " no-shadow";
        bubble.className = className;

        document.getElementById("bubble-icon-preview").textContent = bubbleIcon;
        document.getElementById("bubble-tooltip").textContent = bubbleText;
        document.getElementById("bubble-color-text").value = bubbleColor;
      }
      function updateBubbleColorFromText() {
        const colorText = document.getElementById("bubble-color-text").value;
        if (/^#[0-9A-F]{6}$/i.test(colorText)) {
          document.getElementById("bubble-color").value = colorText;
          updateBubblePreview();
        }
      }
      function setPresetColor(color) {
        document.getElementById("bubble-color").value = color;
        document.getElementById("bubble-color-text").value = color;
        updateBubblePreview();
      }
      function setPosition(position) {
        document.getElementById("bubble-position").value = position;
        document.querySelectorAll(".position-option").forEach((btn) => {
          btn.classList.remove("active");
        });
        document
          .querySelector(`[data-position="${position}"]`)
          .classList.add("active");
        updateBubblePreview();
      }
      function setBubbleSize(size) {
        document.getElementById("bubble-size").value = size;
        updateBubblePreview();
      }
      function updateChatPreview() {
        const chatTitle = document.getElementById("chat-title").value;
        const welcomeMessage = document.getElementById("welcome-message").value;
        const headerColor = document.getElementById("header-color").value;
        const chatTheme = document.getElementById("chat-theme").value;
        const messageStyle = document.getElementById("message-style").value;
        const showBranding = document.getElementById("show-branding").checked;

        document.getElementById("chat-title-preview").textContent = chatTitle;
        document
          .getElementById("welcome-message-preview")
          .querySelector(".message-content").textContent = welcomeMessage;
        document.getElementById("chat-header-preview").style.background =
          headerColor;
        document.getElementById("chat-send-preview").style.background =
          headerColor;
        document.getElementById("header-color-text").value = headerColor;
        document.getElementById("chat-widget-preview").className =
          "chat-widget-preview theme-" + chatTheme;

        const messages = document.querySelectorAll(
          "#chat-messages-preview .bot-message, #chat-messages-preview .user-message",
        );
        messages.forEach((msg) => {
          let baseClass = msg.classList.contains("bot-message")
            ? "bot-message"
            : "user-message";
          msg.className = `${baseClass} style-${messageStyle}`;
        });
        document.getElementById("chat-branding-preview").style.display =
          showBranding ? "block" : "none";
      }
      function updateHeaderColorFromText() {
        const colorText = document.getElementById("header-color-text").value;
        if (/^#[0-9A-F]{6}$/i.test(colorText)) {
          document.getElementById("header-color").value = colorText;
          updateChatPreview();
        }
      }

      // Embed Code & Finalization
      function generateAndDisplayEmbedCode() {
        const widgetId = "qn-" + Date.now();
        let scriptTag = `<script src="http://localhost:8001/embed.js"
    data-quinx-chat-id="${widgetId}"
    data-user="${currentUser.email}"
    data-bubble-color="${document.getElementById("bubble-color").value}"
    data-bubble-icon="${encodeURIComponent(
      document.getElementById("bubble-icon").value,
    )}"
    data-bubble-position="${document.getElementById("bubble-position").value}"
    defer><\/script>`;

        document.getElementById("widget-id-display").textContent = widgetId;
        document
          .getElementById("embed-code-content")
          .querySelector("code").textContent = scriptTag.replace(/^\s+/gm, "");
      }
      function copyEmbedCode() {
        const code = document
          .getElementById("embed-code-content")
          .querySelector("code").textContent;
        navigator.clipboard.writeText(code).then(() => {
          const copyBtn = document.getElementById("copy-btn");
          copyBtn.textContent = "✅ Copied!";
          setTimeout(() => {
            copyBtn.textContent = "📋 Copy Code";
          }, 2000);
        });
      }
      function finishWidgetCreation() {
        const customization = {
          chatTitle: document.getElementById("chat-title").value,
          createdAt: new Date().toISOString(),
        };
        localStorage.setItem(
          "widgetCustomization",
          JSON.stringify(customization),
        );
        completeWidgetCreation();
      }
      function completeWidgetCreation() {
        currentUser.widget_status = "active";
        closeWidgetModal();
        updateWidgetSection();
        updateQuickActions();
        updateDashboardStats();
        alert(
          "🎉 Congratulations!\n\nYour AI chatbot has been created successfully!",
        );
      }
      function logout() {
        localStorage.clear();
        window.location.href = "index.html";
      }

      // Initialize Speech Recognition when page loads
      let speechUI = null;
      document.addEventListener("DOMContentLoaded", function () {
        setTimeout(() => {
          try {
            speechUI = new SpeechUI({
              containerId: "chat-input-container",
              inputId: "chat-input",
              showVisualizer: true,
              showLanguageSelector: true,
              autoSubmit: false,
            });
            console.log("Speech UI initialized successfully");
          } catch (error) {
            console.log("Speech recognition not available:", error.message);
          }
        }, 500);
      });

      // Cleanup on page unload
      window.addEventListener("beforeunload", function () {
        if (speechUI) {
          speechUI.destroy();
        }
      });
    </script>
